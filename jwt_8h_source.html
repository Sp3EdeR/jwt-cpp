342  std::string extract_pubkey_from_cert(const std::string& certstr, const std::string& pw, std::error_code& ec) {
343  ec.clear();
344 #if OPENSSL_VERSION_NUMBER <= 0x10100003L
345  std::unique_ptr<BIO, decltype(&BIO_free_all)> certbio(BIO_new_mem_buf(const_cast<char*>(certstr.data()), certstr.size()), BIO_free_all);
346 #else
347  std::unique_ptr<BIO, decltype(&BIO_free_all)> certbio(BIO_new_mem_buf(certstr.data(), static_cast<int>(certstr.size())), BIO_free_all);
348 #endif
349  std::unique_ptr<BIO, decltype(&BIO_free_all)> keybio(BIO_new(BIO_s_mem()), BIO_free_all);
350  if(!certbio || !keybio) {
351  ec = error::rsa_error::create_mem_bio_failed;
352  return {};
353  }
354 
355  std::unique_ptr<X509, decltype(&X509_free)> cert(PEM_read_bio_X509(certbio.get(), nullptr, nullptr, const_cast<char*>(pw.c_str())), X509_free);
356  if (!cert) {
357  ec = error::rsa_error::cert_load_failed;
358  return {};
359  }
360  std::unique_ptr<EVP_PKEY, decltype(&EVP_PKEY_free)> key(X509_get_pubkey(cert.get()), EVP_PKEY_free);
361  if(!key) {
362  ec = error::rsa_error::get_key_failed;
363  return {};
364  }
365  if(PEM_write_bio_PUBKEY(keybio.get(), key.get()) == 0) {
366  ec = error::rsa_error::write_key_failed;
367  return {};
368  }
369  char* ptr = nullptr;
370  auto len = BIO_get_mem_data(keybio.get(), &ptr);
371  if(len <= 0 || ptr == nullptr) {
372  ec = error::rsa_error::convert_to_pem_failed;
373  return {};
374  }
375  std::string res(ptr, len);
376  return res;
